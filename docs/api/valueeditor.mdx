`<ValueEditor />` may be the default component most commonly [overridden](./querybuilder#valueeditor) with a custom component, but it's quite capable of handling many common scenarios without customization. The query builder below demonstrates each of the available options for the [`Field#valueEditorType` property](../typescript#fields).

:::tip

The [compatibility packages](../compat) also implement these options using appropriate components from the associated style library ([see the demo](https://react-querybuilder.github.io/react-querybuilder/)).

:::

The value actually stored in the rule object within the query is displayed to the right of each value editor. By default, every effort is made to keep the query object serializable. This means that editors that generate non-serializable values (like some date pickers) will store the value as a string, parsing it back into the required object type during the render process. Editors that allow (or require) multiple values, e.g. when the `operator` is "between", will typically store values as comma-separated strings.

_(To highlight the value editors below, all other query builder components have been disabled or hidden.)_

import { QueryBuilderEmbed } from '../../src/components/QueryBuilderEmbed';
import { ValueEditor } from 'react-querybuilder';

<QueryBuilderEmbed
  getOperators={() => [{ name: '=', label: '=' }]}
  controlElements={{
    fieldSelector: ({ value, options }) => (
      <select value={value} disabled>
        {options.map(o => (
          <option key={o.name} value={o.name}>{`valueEditorType: "${o.label}"`}</option>
        ))}
      </select>
    ),
    operatorSelector: ({ value, options }) => (
      <select value={value} disabled>
        {options.map(o => (
          <option key={o.name} value={o.name}>
            {o.label}
          </option>
        ))}
      </select>
    ),
    combinatorSelector: () => null,
    addRuleAction: () => null,
    addGroupAction: () => null,
    removeRuleAction: () => null,
    removeGroupAction: () => null,
    valueEditor: props => (
      <div style={{ display: 'flex' }}>
        <ValueEditor {...props} />
        {'\u00a0'}
        <span
          style={{
            fontSize: 'x-small',
            opacity: 0.5,
          }}>{`(stored value: ${
          typeof props.value === 'boolean' ? props.value : `"${props.value}"`
        })`}</span>
      </div>
    ),
  }}
  fields={[
    { name: 'f1', label: 'text', valueEditorType: 'text' },
    {
      name: 'f2',
      label: 'select',
      valueEditorType: 'select',
      values: [
        { name: 'opt1', label: 'Option 1' },
        { name: 'opt2', label: 'Option 2' },
      ],
    },
    { name: 'f3', label: 'checkbox', valueEditorType: 'checkbox' },
    {
      name: 'f4',
      label: 'radio',
      valueEditorType: 'radio',
      values: [
        { name: 'opt1', label: 'Option 1' },
        { name: 'opt2', label: 'Option 2' },
      ],
    },
    { name: 'f5', label: 'textarea', valueEditorType: 'textarea' },
    {
      name: 'f6',
      label: 'multiselect',
      valueEditorType: 'multiselect',
      values: [
        { name: 'opt1', label: 'Option 1' },
        { name: 'opt2', label: 'Option 2' },
      ],
    },
    { name: 'f7', label: 'date', valueEditorType: 'date' },
    { name: 'f8', label: 'datetime-local', valueEditorType: 'datetime-local' },
    { name: 'f9', label: 'time', valueEditorType: 'time' },
  ]}
  defaultQuery={{
    combinator: 'and',
    rules: [
      { field: 'f1', operator: '=', value: '' },
      { field: 'f2', operator: '=', value: 'opt2' },
      { field: 'f3', operator: '=', value: true },
      { field: 'f4', operator: '=', value: 'opt2' },
      { field: 'f5', operator: '=', value: '' },
      { field: 'f6', operator: '=', value: 'opt1,opt2' },
      { field: 'f7', operator: '=', value: '' },
      { field: 'f8', operator: '=', value: '' },
      { field: 'f9', operator: '=', value: '' },
    ],
  }}
/>
